pipeline {
    agent any

    environment {
        // Repo structure (based on your screenshot)
        SERVICE_DIR = 'smartfreelance-backend/freelancer-service'

        // Docker
        DOCKER_IMAGE = 'nassimrafrafi/smartfreelance'
        DOCKER_CREDENTIALS = 'dockerr'

        // Kubernetes
        KUBE_DEPLOYMENT = 'smartfreelance-deployment'
        KUBE_SERVICE = 'smartfreelance-service'

        // IMPORTANT: Fix Jenkins kube access
        KUBECONFIG = '/var/lib/jenkins/.kube/config'
    }

    stages {

        stage('Checkout') {
            steps {
                git branch: 'nassim-back',
                    url: 'https://github.com/rouamenaa/Esprit-PIDEV-4SAE8-2026-SmartFreelance.git'
            }
        }

        stage('Verify Kubernetes Connection') {
            steps {
                sh '''
                    echo "Checking cluster connection..."
                    kubectl version --client
                    kubectl cluster-info
                    kubectl get nodes
                '''
            }
        }

        stage('Build Docker Image') {
            steps {
                echo 'Building Docker image...'
                dir("${SERVICE_DIR}") {
                    sh '''
                        docker build -t $DOCKER_IMAGE:latest .
                    '''
                }
            }
        }

        stage('Push Docker Image') {
            steps {
                echo 'Pushing Docker image to DockerHub...'
                withCredentials([usernamePassword(
                    credentialsId: "${DOCKER_CREDENTIALS}",
                    usernameVariable: 'DOCKER_USER',
                    passwordVariable: 'DOCKER_PASS'
                )]) {
                    sh '''
                        echo $DOCKER_PASS | docker login -u $DOCKER_USER --password-stdin
                        docker push $DOCKER_IMAGE:latest
                        docker logout
                    '''
                }
            }
        }

        stage('Deploy to Kubernetes (Minikube)') {
            steps {
                echo 'Deploying to Kubernetes...'
                dir("${SERVICE_DIR}") {
                    sh '''
                        echo "Applying Kubernetes manifests..."

                        kubectl apply -f deployment.yaml
                        kubectl apply -f service.yaml

                        echo "Restarting deployment..."
                        kubectl rollout restart deployment/$KUBE_DEPLOYMENT

                        echo "Waiting for rollout..."
                        kubectl rollout status deployment/$KUBE_DEPLOYMENT --timeout=180s
                    '''
                }
            }
        }

        stage('Expose Service URL') {
            steps {
                echo 'Getting Minikube service URL...'
                sh '''
                    minikube service $KUBE_SERVICE --url
                '''
            }
        }
    }

    post {
        success {
            echo '✅ CI/CD pipeline completed successfully!'
        }
        failure {
            echo '❌ Pipeline failed - check Kubernetes or Docker configuration!'
        }
    }
}
