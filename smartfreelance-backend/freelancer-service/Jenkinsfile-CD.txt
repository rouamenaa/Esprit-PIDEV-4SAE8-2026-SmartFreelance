pipeline {
    agent any

    environment {
        SERVICE_DIR = 'smartfreelance-backend/freelancer-service'
        DOCKER_IMAGE = 'nassimrafrafi/smartfreelance'
        DOCKER_CREDENTIALS = 'dockerr'
        KUBE_DEPLOYMENT = 'smartfreelance-deployment'
        KUBE_SERVICE = 'smartfreelance-service'

        // âś… Force Jenkins environment (CRITICAL FIX)
        MINIKUBE_HOME = '/var/lib/jenkins/.minikube'
        KUBECONFIG = '/var/lib/jenkins/.kube/config'
    }

    stages {

        stage('Checkout') {
            steps {
                git branch: 'nassim-back', url: 'https://github.com/rouamenaa/Esprit-PIDEV-4SAE8-2026-SmartFreelance.git'
            }
        }

        stage('Build Image') {
            steps {
                echo 'Building Docker image...'
                dir("${SERVICE_DIR}") {
                    sh 'docker build -t $DOCKER_IMAGE:latest .'
                }
            }
        }

        stage('Push Image') {
            steps {
                echo 'Pushing Docker image...'
                withCredentials([usernamePassword(
                    credentialsId: "${DOCKER_CREDENTIALS}",
                    usernameVariable: 'DOCKER_USER',
                    passwordVariable: 'DOCKER_PASS'
                )]) {
                    sh '''
                        echo $DOCKER_PASS | docker login -u $DOCKER_USER --password-stdin
                        docker push $DOCKER_IMAGE:latest
                        docker logout
                    '''
                }
            }
        }

        stage('Ensure Minikube') {
            steps {
                echo 'Setting up clean Minikube...'
                sh '''
                    export MINIKUBE_HOME=$MINIKUBE_HOME
                    export KUBECONFIG=$KUBECONFIG

                    mkdir -p $MINIKUBE_HOME
                    mkdir -p $(dirname $KUBECONFIG)

                    # đź”Ą Clean old broken configs (VERY IMPORTANT)
                    rm -rf $MINIKUBE_HOME/*
                    rm -rf $(dirname $KUBECONFIG)/*

                    # Start fresh Minikube
                    minikube start --driver=docker

                    kubectl get nodes
                '''
            }
        }

        stage('Deploy to Minikube') {
            steps {
                echo 'Deploying to Kubernetes...'
                dir("${SERVICE_DIR}") {
                    sh '''
                        export MINIKUBE_HOME=$MINIKUBE_HOME
                        export KUBECONFIG=$KUBECONFIG

                        kubectl apply -f k8s/deployment.yaml
                        kubectl apply -f k8s/service.yaml

                        kubectl rollout restart deployment $KUBE_DEPLOYMENT
                        kubectl rollout status deployment/$KUBE_DEPLOYMENT --timeout=180s
                    '''
                }
            }
        }

        stage('Get URL') {
            steps {
                echo 'Getting service URL...'
                sh '''
                    export MINIKUBE_HOME=$MINIKUBE_HOME
                    export KUBECONFIG=$KUBECONFIG

                    minikube service $KUBE_SERVICE --url
                '''
            }
        }
    }

    post {
        success {
            echo 'âś… CD pipeline completed successfully!'
        }
        failure {
            echo 'âťŚ CD pipeline failed!'
        }
    }
}
