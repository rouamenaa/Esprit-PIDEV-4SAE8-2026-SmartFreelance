pipeline {
    agent any

    environment {
        SONARQUBE_ENV = 'sonarr'
        SERVICE_DIR = 'smartfreelance-backend/freelancer-service'
    }

    stages {
        stage('Checkout') {
            steps {
                git branch: 'nassim-back', url: 'https://github.com/rouamenaa/Esprit-PIDEV-4SAE8-2026-SmartFreelance.git'
            }
        }

        stage('Build') {
            steps {
                echo 'Building project...'
                dir("${SERVICE_DIR}") {
                    sh 'mvn clean compile'
                }
            }
        }

        stage('Test') {
            steps {
                echo 'Running tests...'
                dir("${SERVICE_DIR}") {
                    sh 'mvn test'
                }
            }
        }

        stage('SonarQube Analysis') {
            steps {
                echo 'Running SonarQube analysis...'
                withSonarQubeEnv("${SONARQUBE_ENV}") {
                    dir("${SERVICE_DIR}") {
                        sh 'mvn sonar:sonar'
                    }
                }
            }
        }
    }

    post {
        success {
            echo 'Pipeline completed successfully!'

            // Trigger CD pipeline
            build job: 'pi-nassim-cd'
        }

        failure {
            echo 'Pipeline failed!'
        }
    }
}
